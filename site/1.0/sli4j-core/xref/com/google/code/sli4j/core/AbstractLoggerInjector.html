16  package com.google.code.sli4j.core;
17  
18  import java.lang.reflect.Field;
19  import java.lang.reflect.Modifier;
20  
21  import com.google.inject.MembersInjector;
22  
23  /**
24   * The abstract Logger injector implementation, takes care of injecting the
25   * concrete Logger implementation to the logged filed.
26   *
27   * @author Simone Tripodi
28   * @version $Id$
29   */
30  public abstract class AbstractLoggerInjector<L> implements MembersInjector<L> {
31  
32      /**
33       * The logger field has to be injected.
34       */
35      private final Field field;
36  
37      /**
38       * Creates a new Logger injector.
39       *
40       * @param field the logger field has to be injected.
41       */
42      public AbstractLoggerInjector(Field field) {
43          this.field = field;
44      }
45  
46      /**
47       * {@inheritDoc}
48       */
49      public final void injectMembers(Object target) {
50          if (Modifier.isFinal(this.field.getModifiers())) {
51              return;
52          }
53  
54          boolean wasAccessible = this.field.isAccessible();
55          this.field.setAccessible(true);
56          try {
57              if (this.field.get(target) == null) {
58                  this.field.set(target, this.createLogger(this.field.getType()));
59              }
60          } catch (Exception e) {
61              throw new RuntimeException("Impossible to set logger to field '"
62                      + field
63                      + "', see nested exceptions", e);
64          } finally {
65              this.field.setAccessible(wasAccessible);
66          }
67      }
68  
69      /**
70       * Creates a new Logger implementation for the specified Class.
71       *
72       * @return a new Logger implementation.
73       */
74      protected abstract L createLogger(Class<?> klass);
75  
76  }
